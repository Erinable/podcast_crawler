Logging infrastructure for the podcast crawler application.

This module provides a flexible logging system built on top of tracing, supporting multiple output formats and destinations.

use podcast_crawler::infrastructure::{
    config::LoggingConfig,
    logging::init_logger,
};

fn setup_logging() -> Result<(), Box<dyn std::error::Error>> {
    let config = LoggingConfig {
        format: "json".to_string(),
        level: "info".to_string(),
        path: Some("logs".to_string()),
        ..Default::default()
    };
    init_logger(&config)?;
    Ok(())
}
